99#define freetype_gl_error(errno) { \
100 freetype_gl_errno = FTGL_ERR_CAT( FTGL_ERR_PREFIX, errno); \
101 freetype_gl_message = freetype_gl_errstrs[freetype_gl_errno]; \
102 log_error("FTGL Error %s:%d: %s\n", __FILE__, __LINE__, freetype_gl_message); \
103 }
104
105#define freetype_gl_error_str(errno, string) { \
106 freetype_gl_errno = FTGL_ERR_CAT( FTGL_ERR_PREFIX, errno); \
107 freetype_gl_message = freetype_gl_errstrs[freetype_gl_errno]; \
108 log_error("FTGL Error %s:%d: %s '%s'\n", __FILE__, __LINE__, freetype_gl_message, string); \
109 }
110
111#define freetype_gl_warning(errno) { \
112 freetype_gl_errno = FTGL_ERR_CAT( FTGL_ERR_PREFIX, errno); \
113 freetype_gl_message = freetype_gl_errstrs[freetype_gl_errno]; \
114 if(freetype_gl_warnings) log_error("FTGL Warning %s:%d: %s\n", __FILE__, __LINE__, freetype_gl_message); \
115 }
116
117#define freetype_error(errno) { \
118 freetype_gl_errno = errno; \
119 freetype_gl_message = freetype_gl_errstrs[errno]; \
120 log_error("Freetype Error %s:%d: %s\n", __FILE__, __LINE__, freetype_gl_message); \
121 }
